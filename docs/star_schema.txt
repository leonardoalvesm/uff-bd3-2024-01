Schema Design for Yellow Taxi Trip Records

Fact Table

fact_yellow_taxi_trips
    trip_id (Primary Key)
    vendor_id (Foreign Key to dim_vendor)
    pulocation_id (Foreign Key to dim_location)
    dolocation_id (Foreign Key to dim_location)
    rate_code_id (Foreign Key to dim_rate_code)
    payment_type_id (Foreign Key to dim_payment_type)
    tpep_pickup_time_id (Foreign Key to dim_time)
    tpep_dropoff_time_id (Foreign Key to dim_time)
    passenger_count
    trip_distance
    store_and_fwd_flag
    fare_amount
    extra
    mta_tax
    improvement_surcharge
    tip_amount
    tolls_amount
    total_amount
    congestion_surcharge
    airport_fee
    

Dimension Tables
    
dim_vendor
    vendor_id (Primary Key)
    vendor_name

dim_location
    location_id (Primary Key)
    location_latitude
    location_longitude

dim_rate_code
    rate_code_id (Primary Key)
    rate_code_description

dim_payment_type
    payment_type_id (Primary Key)
    payment_type_description

dim_time
    time_id (Primary Key)
    date (Date)
    month (Integer)
    year (Integer)
    day (Integer)
    week (Integer)
    quarter (Integer)
